 19     /**

 21 	 * @param {String} msg message that will be displayed, if error thrown

 37 	 */

 43     /**
 44 	 * @param {Function} func function object that will be called and checked
 45 	 * @param {String} msg message that will be displayed, if error thrown
 46 	 *	 (optional)
 47 	 * @throws AssertionException if the expression not fails
 48 	 */
 49 	meowEngine.assert.fails = function( func, msg ){
 50 
 51 		if( !meowEngine.checks.isFunction( func ) ){
 52 			throw meowEngine.createError("IllegalArgument",
 53                                             "func has to be a function");
 54 		}
 55 
 56 		try{
 57 
 58             func();
 59 		}
 60 		catch( e ){
 61 			/* function thrown an error, assert is correct */
 62 			return;
 63 		}
 64 
 65 		throw meowEngine.createError("AssertionError",msg);
 66 	};

 72     /**
 73 	 * Logger class, to log messages on the console.
 74 	 *
 75 	 * @class
 76 	 */
 77 	meowEngine.test.Logger = 
 78         meowEngine.Class( /** @lends meowEngine.test.Logger# */ {
 79 
 80 		/** @constant */
 81 		LEVEL_OFF : -1,
 82 
 83 		/** @constant */
 84 		LEVEL_INFO : 0,
 85 
 86 		/** @constant */
 87 		LEVEL_FINE : 1,
 88 
 89 		/** @constant */
 90 		LEVEL_WARN : 2,
 91 
 92 		/** @constant */
 93 		LEVEL_CRITICAL : 3,
 94 
 95 		/**
 96 		 * @param {String} identifier will be printed before every log message
 97 		 * @param consoleObject console object, that has the log function
 98 		 */
 99 		constructor : function( identifier, consoleObj )
100 		{
101 			this._identifier = ( typeof identifier === 'string' )?
102 				identifier : "";
103 
104 			/**
105 			 * Status of the logger, the status describes want level of debug
106 			 * messages will be logged. If a message is on a lower level as the
107 			 * set status of the debug logger, then the message won't be
108 			 * displayed on the console.
109 			 *
110 			 * @memberOf meowEngine.test.Logger#
111 			 * @field
112 			 *
113 			 * @example
114 			 * Value can be one of the following constants:
115 			 * meow.Logger.LEVEL_OFF
116 			 * meow.Logger.LEVEL_INFO
117 			 * meow.Logger.LEVEL_FINE
118 			 * meow.Logger.LEVEL_WARN
119 			 * meow.Logger.LEVEL_CRITICAL
120 			 */
121 			this.status = this.LEVEL_INFO;
122 
123 			// set default web console, if possible
124 			if( typeof consoleObj === 'undefined' ){
125 				consoleObj =
126 					( console )? console : ( (Console)? Console : null);
127 			}
128 
129 			/**
130 			 * Default output instance on is Console ( available in Firefox ).
131 			 * If you want to define an own logger, place it as javaScript
132 			 * compatible object in variable 'console' or set it with
133 			 * the constructor of the logger.
134 			 *
135 			 * @example
136 			 * Structure:
137 			 *  void log( String message )
138 			 *
139 			 * @memberOf meowEngine.test.Logger#
140 			 * @field
141 			 * @private
142 			 */
143 			this._logger = consoleObj;
144 		},
145 
146 		/**
147 		 * Logs an information at LEVEL_INFO on the console.
148 		 *
149 		 * @param {String} msg message
150 		 */
151 		info : function( msg )
152 		{
153 			if( this.status >= this.LEVEL_INFO )
154 				this._logger.log( this._identifier+" INFO: "+msg );
155 		},
156 
157 		/**
158 		 * Logs an information at LEVEL_FINE on the console.
159 		 *
160 		 * @param {String} msg message
161 		 */
162 		fine : function( msg )
163 		{
164 			if( this.status >= this.LEVEL_FINE )
165 				this._logger.log( this._identifier+" FINE: "+msg );
166 		},
167 
168 		/**
169 		 * Logs an information at LEVEL_WARN on the console.
170 		 *
171 		 * @param {String} msg message
172 		 */
173 		warn : function( msg )
174 		{
175 			if( this.status >= this.LEVEL_WARN )
176 				this._logger.log( this._identifier+" WARN: "+msg );
177 		},
178 
179 		/**
180 		 * Logs an information at LEVEL_CRITICAL on the console.
181 		 *
182 		 * @param {String} msg message
183 		 */
184 		critical : function( msg )
185 		{
186 			if( this.status >= this.LEVEL_CRITICAL )
187 				this._logger.log( this._identifier+" CRITICAL: "+msg );
188 		}
189 	});

195     var _traceLogger = new meowEngine.test.Logger("TRACER => ");

197     /**
198      * Places tracers around all functions of an object and it's sub objects,
199      * if the recursiv mode is on.
200      *
201      * @param {Object} obj object, that will be traced
202      * @param {Boolean} recursiv if true, all sub objects will be traced too
203      *        (optional)
204      */
205     meowEngine.test.placeTracer = function( obj, recursiv ){
206 
207         if( typeof obj !== 'object' ){
208             throw "placeTracer, obj has to be an object";
209         }
210 
211         var i;
212         var tp; // type of i in object
213         var oF; // old function
214         for( i in obj ){
215 
216             tp = typeof obj[i];
217             if( tp === 'function' && i !== 'toString' ){
218 
219                 oF = obj[i];
220                 obj[i] = _genTraceFunction( i, oF );
221             }
222             else if( recursiv === true && tp === 'object' ){
223                 _placeTr( obj, recursiv );
224             }
225         }
226     }
227 
228     var _genTraceFunction = function( fName, func ){
229 
230         return function(){
231             _traceLogger.info("Enter function "+fName+" of object ("+
232                               this+") with arguments ["+
233                               Array.prototype.slice.call(arguments)+"]");
234 
235             func.apply( this, arguments );
236 
237             _traceLogger.info("Leaving function "+fName+" of object ("+
238                               this+")");
239         };
240     };

243     var _placeTr = meowEngine.test.placeTracer;

249     var _tests = [];

252 	var _SET_UP = "setUp";

255 	var _startsWith = function( str, suffix ){
256 		return str.match("^"+suffix) == suffix;

259     /**
260      * Defines a test case for the meowEngine testing evironment.
261      *
262      * @example
263      * testCase({
264      *
265      *  setUp : function(){
266      *   // code, that will be called first
267      *  },
268      *
269      *  test_Name : function(){
270      *   // a test function
271      *  },
272      *
273      *  tearDown : function(){
274      *   // code, that will be called last
275      *  }
276      * });
277      *
278      * @param {Object} testCase object that contains all tests
279      */
280     meowEngine.test.testCase = function( testCase ){
281 		_tests.push( testCase );
282 	}
283 
284     /**
285      * Runs a test case.
286      *
287      * @param {Object} testCase test case object
288      */
289     meowEngine.test.runTestCase = function( testCase ){
290 
291 		var tp;
292 
293 		_logger.info("Start test case "+testCase._testCase);
294 		var time = (new Date).getTime();
295 		var num=0,numFailed=0;
296 
297 		// setup test case
298 		if( typeof testCase[ _SET_UP ] === 'function' ){
299 			try{
300 				testCase[ _SET_UP ]( _logger );
301 				_logger.info("set up succeed");
302 
303 				for( var f in testCase ){
304 					tp = typeof testCase[f];
305 					if( tp === 'function' ){
306 
307 						if( _startsWith( f , "test_" ) ){
308 							try{
309 								testCase[ f ]( _logger );
310 								_logger.info( f+" succeed");
311 							}
312 							catch( e ){
313 								_logger.info( f+" failed, because of "+e);
314 								numFailed++;
315 							}
316 							finally{
317 								num++;
318 							}
319 						}
320 					}
321 				}
322 
323 				// tear down test case
324 				if( typeof testCase[ _TEAR_DOWN ] === 'function' ){
325 					try{
326 						testCase[ _TEAR_DOWN ]( _logger );
327 						_logger.info("tear down succeed");
328 					}
329 					catch( e ){
330 						_logger.info("tear down failed, because of "+e);
331 					}
332 				}
333 			}
334 			catch( e ){
335 				_logger.info("set up failed, because of "+e);
336 			}
337 		}
338 
339 		time = (new Date).getTime() - time;
340 		_logger.info("Test '"+testCase._testCase+"' done");
341 		_logger.info("Time needed: "+time+" ms");
342 		_logger.info("Number of tests: "+num+", failed: "+numFailed+" \n" );
343 
344 		return ( numFailed == 0 );
345 	};

348      * Runs all registered test cases.
349      */
350 	meowEngine.test.runAll = function(){
351 
352 		var i,e = _tests.length;
353         var _runTest = meowEngine.test.runTestCase;
354 		for( i = 0 ; i < e ; i++ ){
355 			_runTest(_tests[i]);
356 		}
357 	};
