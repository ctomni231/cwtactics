166     /**
167 	 * Creates a new class with a given class implementation
168 	 *
169 	 * @param {Object...} extendClass class(es) that will be extended. Is a
170 	 *							   variable argument, so you can set as many
171 	 *							   classes you want. The first class you def.,
172 	 *							   will be the super class, the other child
173 	 *							   classes of a deeper level. You only can
174 	 *							   access the first class with the Super
175 	 *							   attribute. Parsing algorithm is from the
176 	 *							   outer right subclass, going left to the
177 	 *							   first class argument.
178 	 *							   (should be meow classes)
179 	 * @param {Object} classImpl class implementation
180 	 *
181 	 * @example
182 	 * // simple classes
183 	 * var myClass1 = meow.Class({ x : alert("Hello1") });
184 	 * var myClass2 = meow.Class({ x : alert("Hello2"), y : alert("Bye2") });
185 	 * var myClass3 = meow.Class({ x : alert("Hello3") });
186 	 *
187 	 * // extend classes
188 	 *
189 	 * // myClass4.x => alert("Hello1")
190 	 * // myClass4.Super => myClass1
191 	 * var myClass4 = meow.Class(myClass1,{...});
192 	 *
193 	 * // myClass4.x => alert("Hello1"), myClass4.y => alert("Bye2")
194 	 * // myClass4.Super => myClass1
195 	 * var myClass4 = meow.Class(myClass1,myClass2,{...});
196 	 *
197 	 * @example
198 	 * Fork of http://myjs.fr/my-class/ 04.06.2011
199 	 * Current State: No changes
200 	 *
201 	 * @function
202 	 */
203     meowEngine.Class = function(){
204 
205         var len = arguments.length;
206         var body = arguments[len - 1];
207         var SuperClass = len > 1 ? arguments[0] : null;
208         var hasImplementClasses = len > 2;
209         var Class, SuperClassEmpty;
210 
211         if (body.constructor === Object) {
212             Class = function() {};
213         } else {
214             Class = body.constructor;
215             delete body.constructor;
216         }
217 
218         if (SuperClass) {
219             SuperClassEmpty = function() {};
220             SuperClassEmpty.prototype = SuperClass.prototype;
221             Class.prototype = new SuperClassEmpty();
222             Class.prototype.constructor = Class;
223             Class.Super = SuperClass;
224             extend(Class, SuperClass, false);
225         }
226 
227         if (hasImplementClasses)
228             // for (var i = 1; i < len - 1; i++)
229             // meow parses from right outer subclass, going left, to the first
230             // subclass
231             for (var i = len-2; i > 0; i--)
232                 extend(Class.prototype, arguments[i].prototype, false);
233 
234         extendClass(Class, body);
235 
236         return Class;
237 
238     }

255     var extendClass = function(Class, extension, override) {
256         if (extension.STATIC) {
257             extend(Class, extension.STATIC, override);
258             delete extension.STATIC;
259         }
260         extend(Class.prototype, extension, override)
261     }

263     function extend(obj, extension, override) {
264         var prop;
265         if (override === false) {
266             for (prop in extension)
267                 if (!(prop in obj))
268                     obj[prop] = extension[prop];
269         } else {
270             for (prop in extension)
271                 obj[prop] = extension[prop];
272             if (extension.toString !== Object.prototype.toString)
273                 obj.toString = extension.toString;
274         }
275     }

504 	/**
505 	 * @function
506 	 * @param {Varrags...} args objects, that will be checked
507 	 * @param {Object} obj object where the arguments will searched
508 	 * @return true, if all arguments are in the object, else false
509 	 */
510 	meowEngine.checks.inObject = function(){
511 
512         var len = arguments.length;
513         var obj = arguments[ len-1 ];
514         var i,tmp;
515 
516         if( !isObject(obj) ){
517             throw "inObject: last parameter must be an object";
518         }
519 
520         // use a direct check, if only one argument to check
521         if( len == 2 ){
522 
523             tmp = arguments[0];
524 
525             for( i in obj ){
526                 if( obj[i] === tmp )
527                     return true;
528             }
529 
530             return false;
531         }
532         // if many arguments present, use a cached check
533         else{
534 
535             var objList = new Array();
536             for( i in obj ){
537                 objList.push( obj[i] );
538             }
539 
540             var e2 = len-1;
541             for( i = 0; i<e2; i++){
542                 if( objList.indexOf( arguments[i] ) == -1 ) return false;
543             }
544 
545             return true;
546         }
547     };

791 	meowEngine.Signal = meowEngine.Class( /** @lends meowEngine.Signal# */ {
792 
793         /** */
794         constructor : function(){
795 
796             /**
797              * @private
798              * @lends meowEngine.Signal#
799              */
800             this._data = [];
801         },
802 
803         /**
804          * Filters an argument with the listener functions.
805          *
806          * @param {Boolean} ignoreBreak if true, a return true of a filter, that
807          *							    indicates a break of the filter chain,
808          *							    will be ignored
809          */
810 		dispatch : function( ignoreBreak, argList ){
811 
812             var list = this._data;
813             var e = list.length;
814             var res;
815 
816             // invoke argument on all filter objects
817             for( var i = 0 ; i < e ; i++ )
818             {
819                 res = list[i].apply(null,argList);
820 
821                 // if no ignore on break and filter returns true,
822                 // then the chain will be broken
823                 if( ignoreBreak === false && res === true )
824                     break;
825             }
826 
827             list = null
828         },
829 
830         /**
831          * Adds a filter to the filter chain. The function get one argument by
832          * the filter chain, if a filter process happens. The function can
833          * return true, if the complete filter process should be breaked.
834          *
835          * @param {Function} func listener function
836          * @param {Boolean} atFirst if true, the filter will appended at the
837          *							head of the filter chain
838          */
839 		add : function( func , atFirst ){
840             ( atFirst === true )? this._data.unshift( func ) :
841                                   this._data.push( func );
842         },
843 
844         /**
845          * Removes a listener from the signal.
846          *
847          * @param {Function} func listener function
848          * @return true, if the function is removed, else false ( if e.g. the
849          *		   function does not exists in the signal )
850          */
851 		remove : function( func ){
852 
853             var i = this._data.indexOf( func );
854             if( i != -1 ){
855 
856                 this._data.splice( i , 1 );
857                 return true;
858             }
859             else{
860                 return false;
861             }
862         }
863 	});

892

894 			throw "MeowEngine onSignal, illegal argument(s) given";
895 		}
896 
897 		// exists signal already? If not, create it
898 		if( typeof _signals[ sigName ] === 'undefined' ){
899 			_signals[ sigName ] = new meowEngine.Signal();
900 		}
901 
902 		_signals[ sigName ].add( listener );
903 	}
