  7 /**
  8  * Checks if an unit has targets for a given weapon tag.
  9  *
 10  * @param unit attacking unit object
 11  * @param wpTag mainWeapon|subWeapon
 12  * @param x x coordinate (optional)
 13  * @param y y coordinate (optional)
 14  * @return true|false
 15  */
 16 game.hasTargets = function( unit, wpTag, x, y, moved ){
 17   if( arguments.length === 2 ){
 18     x = unit.x;
 19     y = unit.y;
 20   }
 21 
 22   var spid = unit.owner;
 23   var steam = game.player( unit.owner ).team;
 24 
 25   var usheet = game.unitSheet( unit.type );
 26   var wp     = game.weaponSheet(
 27     ( wpTag === game.PRIMARY_WEAPON_TAG )?
 28       usheet[game.PRIMARY_WEAPON_TAG] : usheet[game.SECONDARY_WEAPON_TAG]
 29   );
 30   
 31   if( wp === undefined ) return false;
 32 
 33   var minR = wp.minRange;
 34   var maxR = wp.maxRange;
 35 
 36   var lX;
 37   var hX;
 38   var lY = y-maxR;
 39   var hY = y+maxR;
 40   if( lY < 0 ) lY = 0;
 41   if( hY >= domain.mapHeight ) hY = domain.mapHeight-1;
 42   for( ; lY<=hY; lY++ ){
 43 
 44     var disY = Math.abs( lY-y );
 45     lX = x-maxR+disY;
 46     hX = x+maxR-disY;
 47     if( lX < 0 ) lX = 0;
 48     if( hX >= domain.mapWidth ) hX = domain.mapWidth-1;
 49     for( ; lX<=hX; lX++ ){
 50 
 51       if( game.distance( x,y, lX,lY ) >= minR ){
 52         var tUnit = game.unitByPos( lX, lY );
 53         if( tUnit !== null && tUnit.owner !== spid &&
 54           game.player(tUnit.owner).team !== steam ){
 55 
 56           var dmg = game.getBaseDamage( wp, tUnit.type );
 57           if( dmg > 0 ){
 58 
 59             return true;
 60           }
 61         }
 62       }
 63     }
 64   }
 65 };

 67 /**
 68  * Invokes a battle between an attacker against a defender. If the defender
 69  * weapon is defined and the defender is alive after the attack then it will
 70  * start a counter attack if the weapons suits for a counter attack.
 71  *
 72  * @param attid
 73  * @param defid
 74  * @param attDmg
 75  * @param defDmg
 76  * @param attwpid
 77  * @param defwpid
 78  */
 79 game.unitAttack = function( attid, defid, attDmg, defDmg, attwp, defwp ) {
 80 
 81   var att = game.unitById( attid );
 82   var def = game.unitById( defid );
 83 
 84   def.hp -= attDmg;
 85 
 86   // decrease ammo
 87   if( attwp.useAmmo !== -1 ){
 88     att.ammo--;
 89   }
 90 
 91   // fill co power meter
 92   game.increasePowerMeter( att.owner, 0.5*0 );
 93   game.increasePowerMeter( def.owner, 0 );
 94 
 95   if( def.hp < 0 ){
 96 
 97     // defender destroyed
 98     cwt.destroyUnit( defid );
 99   }
100   else if( defwp !== null ){
101 
102     // counterattack
103     att.hp -= defDmg;
104 
105     // decrease ammo
106     if( defwp.useAmmo !== -1 ){
107       def.ammo--;
108     }
109 
110     // fill co power meter
111     cwt.increasePowerMeter( att.owner, 0 );
112     cwt.increasePowerMeter( def.owner, 0.5*0 );
113 
114     if( att.hp < 0 ){
115 
116       // attacker destroyed
117       cwt.destroyUnit( attid );
118     }
119   }
120 };

126  * @param uType {string} unit type

128 game.getBaseDamage = function( weapon, uType ){
129 
130   var dmg;
131 
132   dmg = weapon.damages[ uType ];
133   if( dmg !== undefined ) return dmg;
134 
135   dmg = weapon.damages[ "*" ];
136   if( dmg !== undefined ) return dmg;
137 
138   return 0;
139 };

219       hX = tx+maxR-disY;

225           var tUnit = game.unitByPos( lX, lY );
226           if( tUnit !== null && tUnit.owner !== spid &&

247       ( game.primaryWeaponOfUnit(selected) !== null &&

278 /********
279 
280    D%=[B*ACO/100+R]*(AHP/10)*[(200-(DCO+DTR*DHP))/100]
281 
282    D = Actual damage expressed as a percentage
283 
284    B = Base damage (in damage chart)
285    ACO = Attacking CO attack value (example:130 for Kanbei)
286 
287    R = Random number 0-9
288 
289    AHP = HP of attacker
290 
291    DCO = Defending CO defense value (example:80 for Grimm)
292    DTR = Defending Terrain Stars (IE plain = 1, wood = 2)
293    DHP = HP of defender
294 
295  Denoted as xxxXXXX where x = small star and X = big star..for now.
296  Every star is worth 9000 fund at the start of the game. Each additional
297  use of CO Power(including SCOP) increase the value of each star by 1800
298  fund up to the tenth usage, when it won't increase any further.
299 
300  Stars on your Charge Meter can be charged in two ways:
301 
302  Damaging your oppoent's units. You gain meter equal to half the fund damage
303  you deal. Receiving damage from your opponent. You gain meter equal to the
304  fund damage you take. Keep in mind that AWBW only keeps track of real numbers
305  for the purpose of Charge Meter calculation. That means taking a 57% attack
306  and ending up with 5 hp only adds 0.5*full cost of unit to your Charge Meter.
307  In Summary, the amount of charge added to your meter can be calculated as:
308 
309    (0.5*0.1*Damage Dealt*Cost of Unit X)+(0.1*Damage Received*Cost of Unit Y)
310 
311 
312  It should be noted that a COs meter does not charge during the turn they activate a power.
313 
314  *****************/
