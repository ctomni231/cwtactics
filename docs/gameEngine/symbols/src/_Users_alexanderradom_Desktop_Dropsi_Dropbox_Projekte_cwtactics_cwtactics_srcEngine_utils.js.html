1 /**
  2  * Fills an array with a value. Works also for matrix objects.
  3  *
  4  * @param arr
  5  * @param defaultValue
  6  */
  7 util.fill = function( arr, defaultValue ){
  8   var isFN = typeof defaultValue === 'function';
  9 
 10   if( arr.__matrixArray__ === true ){
 11 
 12     // COMPLEX ARRAY (MATRIX) OBJECT
 13     for(var i = 0, e = arr.length; i < e; i++){
 14       for(var j = 0, ej = arr.length; j < ej; j++){
 15         if( isFN ) arr[i][j] = defaultValue( i,j );
 16         else       arr[i][j] = defaultValue;
 17       }
 18     }
 19   }
 20   else{
 21 
 22     // SIMPLE ARRAY OBJECT
 23     for(var i = 0, e = arr.length; i < e; i++){
 24       if( isFN ) arr[i] = defaultValue( i );
 25       else       arr[i] = defaultValue;
 26     }
 27   }
 28 };

 30 /**
 31  * Creates a list and fills it with default values.
 32  *
 33  * @param len
 34  * @param defaultValue
 35  */
 36 util.list = function( len, defaultValue ){
 37   if( defaultValue === undefined ){ defaultValue = null; }
 38 
 39   var isFN = typeof defaultValue === 'function';
 40   var warr = [];
 41   for (var i = 0; i < len; i++) {
 42     if( isFN ) warr[i] = defaultValue( i );
 43     else       warr[i] = defaultValue;
 44   }
 45 
 46   return warr;
 47 };

 49 /**
 50  * Creates a matrix (table) and fills it with default values.
 51  *
 52  * @param w
 53  * @param h
 54  * @param defaultValue
 55  */
 56 util.matrix = function( w, h, defaultValue ){
 57 
 58   if( defaultValue === undefined ){ defaultValue = null; }
 59 
 60   var isFN = typeof defaultValue === 'function';
 61   var warr = [];
 62   for (var i = 0; i < w; i++) {
 63 
 64     warr[i] = [];
 65     for (var j = 0; j < h; j++) {
 66 
 67       if( isFN ) warr[i][j] = defaultValue( i,j );
 68       else       warr[i][j] = defaultValue;
 69     }
 70   }
 71 
 72   // meta data
 73   warr.__matrixArray__ = true;
 74 
 75   return warr;
 76 };

 90 /** @config */
 91 util.logWriter = function( level, string ){
 92   switch( level ){
 93 
 94     case util.LOG_ERROR:
 95       console.error( string );
 96       break;
 97 
 98     case util.LOG_INFO:
 99       console.log( string );
100       break;
101 
102     case util.LOG_WARN:
103       console.warn( string );
104       break;
105 
106     default:  console.log( string );
107   }
108 };
