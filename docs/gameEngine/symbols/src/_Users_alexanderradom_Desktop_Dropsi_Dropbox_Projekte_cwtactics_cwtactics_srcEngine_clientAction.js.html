 64 /**
 65  * Pushes a message into the message bugger.
 66  *
 67  * @param msg message content as js object
 68  * @private
 69  */
 70 controller._pushMsgToBuffer = function (msg) {
 71   if (controller._bufferData[ controller._bufferWriteIndex ] !== null) {
 72     util.logError("message buffer is full");
 73     return;
 74   }
 75 
 76   if (util.DEBUG) {
 77     util.logInfo("adding message (", util.objectToJSON(msg), ") to buffer");
 78   }
 79 
 80   // WRITE MSG AND INCREASE COUNTER
 81   controller._bufferData[ controller._bufferWriteIndex ] = msg;
 82   controller._bufferWriteIndex++;
 83   if (controller._bufferWriteIndex === CWT_MAX_BUFFER_SIZE) {
 84     controller._bufferWriteIndex = 0;
 85   }
 86 };
 87 
 88 /**
 89  * Returns true if the buffer is not empty else false.
 90  *
 91  * @private
 92  */
 93 controller._isMsgInBuffer = function () {
 94   return ( controller._bufferData[ controller._bufferReadIndex ] !== null );
 95 };
 96 
 97 /**
 98  * Pops a message from the message buffer and returns its content.
 99  *
100  * @return message content
101  * @private
102  */
103 controller._popMsgFromBuffer = function () {
104   if (controller._bufferData[ controller._bufferReadIndex ] === null) {
105     util.logError("message buffer is empty");
106   }
107   var msg = controller._bufferData[ controller._bufferReadIndex ];
108 
109   // INCREASE COUNTER AND FREE SPACE
110   controller._bufferData[ controller._bufferReadIndex ] = null;
111   controller._bufferReadIndex++;
112   if (controller._bufferReadIndex === CWT_MAX_BUFFER_SIZE) {
113     controller._bufferReadIndex = 0;
114   }
115 
116   return msg;
117 };

132 /**
133  * Clears the selection map for a new selection process.
134  *
135  * @param centerX
136  * @param centerY
137  */
138 controller._clearSelection = function (centerX, centerY) {
139 
140   var e = CWT_MAX_SELECTION_RANGE * 2 + 1;
141   for (var x = 0; x < e; x++) {
142     for (var y = 0; y < e; y++) {
143       controller.selectionTargets[x][y] = -1;
144     }
145   }
146 
147   // right bounds are not important
148   centerX = Math.max(0, centerX - CWT_MAX_SELECTION_RANGE);
149   centerY = Math.max(0, centerY - CWT_MAX_SELECTION_RANGE);
150 
151   controller._selectionScreenX = centerX;
152   controller._selectionScreenY = centerY;
153 };
154 
155 /**
156  * Gets the value of a selection position.
157  *
158  * @param x
159  * @param y
160  */
161 controller.valueOfSelectionPos = function (x, y) {
162   var sx = x - controller._selectionScreenX;
163   var sy = y - controller._selectionScreenY;
164   var maxLen = 2 * CWT_MAX_SELECTION_RANGE + 1;
165 
166   if (sx < 0 || sy < 0 || sx >= maxLen || sy >= maxLen) {
167     return -1;
168   }
169   else return controller.selectionTargets[sx][sy];
170 };
171 
172 /**
173  * Sets the value of a selection position.
174  *
175  * @param x
176  * @param y
177  * @param value (default=1)
178  */
179 controller.setValueOfSelectionPos = function (x, y, value) {
180   var sx = x - controller._selectionScreenX;
181   var sy = y - controller._selectionScreenY;
182   var len = CWT_MAX_SELECTION_RANGE * 2 + 1;
183   if (value === undefined) value = 1;
184 
185   if (sx < 0 || sx >= len || sy < 0 || sy >= len) {
186     util.logError("position (", x, ",", y, ") is out of move map bounds");
187   }
188   else controller.selectionTargets[sx][sy] = value;
189 };

259 /**
260  * Returns the shortest path from the selected position to a given position.
261  * The used algorithm is a-star.
262  *
263  * @param tx
264  * @param ty
265  */
266 controller.returnPath = function (tx, ty) {
267   var stx = controller.selectedSourceX;
268   var sty = controller.selectedSourceY;
269 
270   if (util.DEBUG) util.logInfo(
271     "searching path from (", stx, ",", sty, ") to (", tx, ",", ty, ")"
272   );
273 
274   // var graph = new Graph( nodes );
275   var graph = new Graph(controller.selectionTargets);
276 
277   var start = graph.nodes[stx - controller._selectionScreenX][sty -
278     controller._selectionScreenY];
279 
280   var end = graph.nodes[tx - controller._selectionScreenX][ty -
281     controller._selectionScreenY];
282 
283   var path = astar.search(graph.nodes, start, end);
284 
285   if (util.DEBUG) util.logInfo("calculated way is", path);
286 
287   var codesPath = [];
288   var cx = stx - controller._selectionScreenX;
289   var cy = sty - controller._selectionScreenY;
290   var cNode;
291 
292   for (var i = 0, e = path.length; i < e; i++) {
293     cNode = path[i];
294 
295     var dir;
296     if (cNode.x > cx) dir = 1;
297     if (cNode.x < cx) dir = 3;
298     if (cNode.y > cy) dir = 2;
299     if (cNode.y < cy) dir = 0;
300 
301     if (dir === undefined) throw Error();
302 
303     codesPath.push(dir);
304 
305     cx = cNode.x;
306     cy = cNode.y;
307   }
308 
309   return codesPath;
310 };

312 /**
313  * Generates the action list for the current selected tile ( or object if
314  * tile is a property or occupied by an unit ).
315  *
316  * @private
317  */
318 controller._generateActionList = function () {
319   var list = controller.actionList;
320 
321   var actions = Object.keys( window.actions );
322   var property = game.propertyByPos(
323     controller.selectedTargetX, controller.selectedTargetY);
324 
325   var unit = game.unitByPos(
326     controller.selectedTargetX, controller.selectedTargetY);
327 
328   var selectedUnit = (controller.selectedObject === -1)? 
329                       null: game.unitById(controller.selectedObject);
330                       
331   var rStP = game.relationshipBetween(selectedUnit, property);
332   var rStU = game.relationshipBetween(selectedUnit, unit);
333 
334   for (var i = 0, e = actions.length; i < e; i++) {
335     var key = actions[i];
336 
337     // x,y,rStP,rStU,selected,property,unit
338     if ( window.actions[key].condition(
339       controller.selectedTargetX, controller.selectedTargetY,
340       rStP, rStU,
341       selectedUnit,
342       property, unit)
343       ) {
344       list.push(key);
345     }
346   }
347 };

349 /**
350  * Flushes the current saved user action data into action command objects.
351  *
352  * @private
353  */
354 controller._flushAction = function () {
355   // TODO --> SHARE CALL
356 
357   var property = game.propertyByPos(
358     controller.selectedTargetX, controller.selectedTargetY);
359 
360   var unit = game.unitByPos(
361     controller.selectedTargetX, controller.selectedTargetY);
362 
363   // DO MOVE
364   if (controller.selectedMovePath !== null &&
365     controller.selectedMovePath.length > 0) {
366 
367     controller._pushMsgToBuffer({
368       k:"move",
369       a:[
370         controller.selectedObject,
371         controller.selectedSourceX, controller.selectedSourceY,
372         controller.selectedMovePath
373       ]
374     });
375   }
376 
377   // DO ACTION
378   controller._pushMsgToBuffer({
379     k:controller.selectedAction,
380     a:[
381       controller.selectedTargetX, controller.selectedTargetY,
382       controller.selectedObject,
383       ( property === null ) ? -1 : game.extractPropertyId(property),
384       ( unit === null ) ? -1 : game.extractUnitId(unit),
385       controller.selectedActionTargetX, controller.selectedActionTargetY,
386       controller.selectedSubMenuAction
387     ]
388   });
389 
390   controller.reset();
391 };

393 /**

395  */
396 controller.reset = function () {
397   if (util.DEBUG) util.logInfo("reset action API to clean state");
398 
399   controller.currentState = controller.STATE_IDLE;
400   controller.selectedSourceX = -1;
401   controller.selectedSourceY = -1;
402   controller.selectedTargetX = -1;
403   controller.selectedTargetY = -1;
404   controller.selectedActionTargetX = -1;

406   controller.selectedObject = -1;
407   controller.selectedMovePath = null;
408   controller.selectedAction = null;
409   controller.selectedSubMenuAction = null;
410 
411   controller.subActionList.splice(0);
412   controller.actionList.splice(0);
413 };

415 /**
416  * Selects a tile by a position. If the tile is a property or occupied by
417  * an own unit, the move path selection state will be chosen and the
418  * selected object will be set as selected data.
419  *
420  * @param x
421  * @param y
422  */
423 controller.tickTile = function (x, y) {
424 
425   var pid = game.getTurnOwner();
426   switch (controller.currentState) {
427 
428     // -----------------------------------------------------
429     case controller.STATE_IDLE :
430       controller.selectedSourceX = x;
431       controller.selectedSourceY = y;
432 
433       var selected = -1;
434       var state = controller.STATE_SELECT_ACTION;
435 
436       var unit = game.unitByPos(x, y);
437       if (unit !== null && unit.owner === pid &&
438         game.canAct(game.extractUnitId(unit))) {
439 
440         controller.selectedObject = game.extractUnitId(unit);
441         controller.currentState = controller.STATE_SELECT_MOVE_PATH;
442         controller._clearSelection(x, y);
443         game.prepareMovepathSelection( controller.selectedObject,
444           controller.valueOfSelectionPos, controller.setValueOfSelectionPos );
445       }
446       else {
447 
448         controller.selectedTargetX = x;
449         controller.selectedTargetY = y;
450         controller.currentState = controller.STATE_SELECT_ACTION;
451         controller._generateActionList();
452       }
453 
454       break;
455 
456     // -----------------------------------------------------
457     case controller.STATE_SELECT_ACTION_TARGET :
458       controller.selectedActionTargetX = x;
459       controller.selectedActionTargetY = y;
460       controller._flushAction();
461       break;
462 
463     // -----------------------------------------------------
464     case controller.STATE_SELECT_ACTION_FREE_TARGET :
465       controller.selectedActionTargetX = x;
466       controller.selectedActionTargetY = y;
467       controller._flushAction();
468       break;
469 
470     // -----------------------------------------------------
471     case controller.STATE_SELECT_MOVE_PATH :
472       if (util.DEBUG) {
473         util.logInfo("target pos selected, generating path by a star");
474       }
475 
476       controller.selectedTargetX = x;
477       controller.selectedTargetY = y;
478 
479       // MOVE BY POS MEANS A PATH MUST BE GENERATED
480       if (controller.selectedMovePath === null) {
481         controller.selectedMovePath = controller.returnPath(x, y);
482       }
483 
484       controller._generateActionList();
485       controller.currentState = controller.STATE_SELECT_ACTION;
486 
487       if (util.DEBUG) {
488         util.logInfo(
489           "got path", controller.selectedMovePath,
490           "with target pos (", x, ",", y, ")"
491         );
492       }
493       break;
494 
495     // -----------------------------------------------------
496     default:
497       util.logError("cannot select tile in current active state");
498   }
499 };

501 /**
502  * Sets a move path.
503  *
504  * @param movepath
505  */
506 controller.tickPath = function (movepath) {
507   if (util.DEBUG) {
508     util.logInfo("path selected, calculating target position");
509   }
510 
511   if (controller.currentState !== controller.STATE_SELECT_MOVE_PATH) {
512     util.logError("cannot set path because action API is not in move state");
513   }
514 
515   var tx = controller.selectedSourceX;
516   var ty = controller.selectedSourceY;
517 
518   for (var i = 0, e = movepath.length; i < e; i++) {
519     switch (movepath[i]) {
520 
521       case game.MOVE_CODE_DOWN :
522         ty++;
523         break;
524       case game.MOVE_CODE_RIGHT :
525         tx++;
526         break;
527       case game.MOVE_CODE_UP :
528         ty--;
529         break;
530       case game.MOVE_CODE_LEFT :
531         tx--;
532         break;
533 
534       default:
535         util.logError("unknown move code", controller.selectedMovePath[i]);
536         throw Error();
537     }
538   }
539 
540   // SET TARGET BY PATH
541   controller.selectedMovePath = movepath;
542   controller.selectedTargetX = tx;
543   controller.selectedTargetY = ty;
544 
545   if (util.DEBUG) {
546     util.logInfo("got path", movepath, "with target pos (", tx, ",", ty, ")");
547   }
548 };

550 /**
551  * Selects an action key.
552  *
553  * @param actionKey
554  */
555 controller.selectEntry = function (actionKey) {
556   var state = controller.currentState;
557   var actionState = ( state === controller.STATE_SELECT_ACTION );
558   var subActionState = ( state === controller.STATE_SELECT_SUBMENU_ACTION );
559   var selected = controller.selectedObject;
560 
561   if (actionState) {
562     controller.selectedAction = actionKey;
563 
564     // BUILD SUB MENU BY ACTION CALL
565     if (actions[actionKey].hasSubMenu === true) {
566       controller.currentState = controller.STATE_SELECT_SUBMENU_ACTION;
567       actions[actionKey].prepareSubMenu(
568         controller.subActionList,
569         (selected !== -1) ? game.unitById(selected) : null,
570         controller.selectedTargetX, controller.selectedTargetY
571       );
572     }
573     // SELECT TARGET BY ACTION
574     else if (actions[actionKey].targetSelection === true) {
575 
576       if( actions[actionKey].freeTarget === true ){
577         controller.currentState = controller.STATE_SELECT_ACTION_FREE_TARGET;
578       }
579       else{
580 
581         controller.currentState = controller.STATE_SELECT_ACTION_TARGET;
582         controller._clearSelection(
583           controller.selectedTargetX, controller.selectedTargetY);
584 
585         actions[actionKey].prepareTargets(
586           (selected !== -1) ? game.unitById(selected) : null,
587           controller.selectedTargetX, controller.selectedTargetY,
588           controller.selectedSubMenuAction
589         );
590       }
591     }
592     // DO ACTION BY ENTRY SELECTION
593     else controller._flushAction();
594   }
595   else if (subActionState) {
596     controller.selectedSubMenuAction = actionKey;
597 
598     if (actions[ controller.selectedAction ].targetSelection === true) {
599       if( actions[ controller.selectedAction ].freeTarget === true ){
600         controller.currentState = controller.STATE_SELECT_ACTION_FREE_TARGET;
601       }
602       else{
603 
604         controller.currentState = controller.STATE_SELECT_ACTION_TARGET;
605         controller._clearSelection(
606           controller.selectedTargetX, controller.selectedTargetY);
607 
608         actions[ controller.selectedAction ].prepareTargets(
609           (selected !== -1) ? game.unitById(selected) : null,
610           controller.selectedTargetX, controller.selectedTargetY,
611           controller.selectedSubMenuAction
612         );
613       }
614     }
615     // DO ACTION AFTER SUB ENTRY SELECTION
616     else controller._flushAction();
617   }
618 };

628 /**
629  * Pops and evaluates the next available action object from the user action
630  * stack.
631  *
632  * @return action key if an action was in the stack, else null
633  */
634 controller.evaluateNextAction = function () {
635   // TODO move out as event
636 
637   // NO ACTION AVAILABLE --> NO ACTION EVALUATED
638   if (!controller.hasNextAction()) return null;
639 
640   var msg = controller._popMsgFromBuffer();
641 
642   if (msg.k === "move") {
643 
644     // DO MOVE
645     game.moveUnitFromAtoB.apply(null, msg.a);
646     return msg;
647   }
648   else {
649 
650     // DO ACTION
651     actions[ msg.k ].action.apply(null, msg.a);
652     return msg;
653   }
654 };
