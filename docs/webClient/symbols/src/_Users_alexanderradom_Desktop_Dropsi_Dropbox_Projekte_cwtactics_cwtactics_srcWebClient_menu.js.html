  4 menu.triggerMenuCancel = function(){
  5   if( util.DEBUG ) util.logInfo("closing menu via cancel event");
  6

  8   $( '#'+client.ID_MENU ).hide();

 12 /**
 13  *
 14  */
 15 menu.triggerMenuAction = function( index ){
 16   if( util.DEBUG ) util.logInfo("doing menu action for index", index);
 17 
 18   sound.play("MENU_OK");
 19   $( '#'+client.ID_MENU ).hide();
 20   userInput.invokeActionFromMenu( index );
 21 };
 22 
 23 /**
 24  *
 25  */
 26 menu._connectListener = function( el, index ){
 27   el.off('click');
 28   el.click(function(){
 29     menu.triggerMenuAction( index );
 30   });
 31 };

 33 /**
 34  * @private
 35  */
 36 menu.showMenu = function( x, y, actions ){
 37 
 38   if( x === -1 || y === -1 ){
 39     x = menu._oldMenuX;
 40     y = menu._oldMenuY;
 41   }
 42   else{
 43     // GET SCREEN POSITIONS
 44     x = x - screen.screenX;
 45     y = y - screen.screenY;
 46     if( y > (screen.screenHeight/2) ) y = y - actions.length - 1 ;
 47     if( x > (screen.screenWidth/2) ) x = x - 6;
 48   }
 49 
 50   if( util.DEBUG ) util.logInfo("open action menu");
 51 
 52   // GENERATE ENTRIES
 53 
 54   for( var i=0,e=9 ; i<=e; i++ ){ $('#menuEntry_'+i).hide(); }
 55   for( var i=0,e=actions.length ; i<e; i++ ){
 56     $('#menuEntry_'+i).show();
 57     menu._connectListener( $('#menuEntry_'+i), i );
 58     $('#menuEntry_'+i).html( locale.localizedString( actions[i] ) );
 59   }
 60 
 61   $('#menuEntry_CANCEL').click( function(){ menu.triggerMenuCancel();});
 62   $('#menuEntry_CANCEL').html( locale.localizedString( "cancel" ) );
 63 
 64   menu._oldMenuX = x;
 65   menu._oldMenuY = y;
 66 
 67   $( '#'+client.ID_MENU ).css("top",y*32+16);
 68   $( '#'+client.ID_MENU ).css("left",x*32+16);
 69   $( '#'+client.ID_MENU ).css("z-index",2000);
 70   $( '#'+client.ID_MENU ).show( );
 71 };
 72

 74  * Holds the last known x coordinate of the menu.
 75  *
 76  * @private
 77  */

 81  * Holds the last known y coordinate of the menu.
 82  *
 83  * @private
 84  */
