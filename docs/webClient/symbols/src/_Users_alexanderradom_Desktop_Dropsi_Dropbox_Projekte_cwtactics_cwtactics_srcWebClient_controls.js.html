 58 /**
 59  * Resets the current active move path for a given unit.
 60  *
 61  * @param uid
 62  */
 63 userInput.resetCurrentMovePath = function( uid ){
 64   util.fill( userInput.currentMovePath, null );
 65 
 66   var unit = game.unitById( uid );
 67   userInput.currentMovePathSourceX = unit.x;
 68   userInput.currentMovePathSourceY = unit.y;
 69   userInput.currentMovePathX = unit.x;
 70   userInput.currentMovePathY = unit.y;
 71   userInput.currentMoveIndex = 0;
 72 
 73   userInput.currentMoveLeftPoints = Math.min( unit.fuel,
 74     game.unitSheet( unit.type ).moveRange );
 75 };

 77 /**
 78  * Returns a valid path array from the current active move path selection.

 81   var path = [];
 82 
 83   for( var i=0; i<CWT_MAX_MOVE_RANGE; i++ ){
 84     if( userInput.currentMovePath[i] === null ) break;

 91 /**
 92  * Invokes an action on a given position.
 93  *
 94  * @param x
 95  * @param y
 96  */
 97 userInput.invokeActionAt = function( x,y ){
 98   sound.play("MENU_OK");
 99 
100   // CHECK REDRAWS OF THE FOCUS TILES --> BEFORE
101   for(var ix=screen.screenX,xe=screen.screenWidth; ix<xe; ix++){
102     for(var iy=screen.screenY,ye=screen.screenHeight; iy<ye; iy++){
103       if( controller.valueOfSelectionPos(ix,iy) > 0 ){
104         screen.drawnMap[ix][iy] = true;
105         screen.drawChanges++;
106       }
107     }
108   }
109 
110   if( controller.currentState === controller.STATE_SELECT_ACTION ) return;
111   if( controller.currentState === controller.STATE_SELECT_MOVE_PATH ){
112     if( userInput.currentMoveIndex > 0 ){
113       controller.tickPath( userInput.getMovePathFromSelection() );
114     }
115   }
116 
117   if( util.DEBUG ) util.logInfo("got action request for (",x,",",y,")");
118   controller.tickTile(x,y);
119 
120   if( controller.currentState === controller.STATE_SELECT_ACTION ){
121     menu.showMenu( x,y, controller.actionList );
122   }
123   else if( controller.currentState === controller.STATE_SELECT_MOVE_PATH ){
124     userInput.resetCurrentMovePath( controller.selectedObject );
125   }
126 };

159 /**
160  * Shifts the screen in relationship to the screen movement variables.
161  */
162 userInput.solveMapShift = function(){
163   var dir = -1;
164 
165   if( userInput.moveScreenX !== 0 ){
166 
167     if( userInput.moveScreenX < 0 ){ dir = 3; userInput.moveScreenX++; }
168     else                           { dir = 1; userInput.moveScreenX--; }
169   }
170   else if( userInput.moveScreenY !== 0 ){
171 
172     if( userInput.moveScreenY < 0 ){ dir = 0; userInput.moveScreenY++; }
173     else                           { dir = 2; userInput.moveScreenY--; }
174   }
175 
176   if( dir !== -1 ) userInput.mapShift( dir, 1 );
177 };

179 /**
180  * New map shift method. This one uses the update step based screen
181  * movement algorithm.
182  */
183 userInput.betterMapShift = function( dir, len ){
184 
185        if( dir === 0 ) userInput.moveScreenY -= len;
186   else if( dir === 1 ) userInput.moveScreenX += len;
187   else if( dir === 2 ) userInput.moveScreenY += len;
188   else if( dir === 3 ) userInput.moveScreenX -= len;
189 };

191 /** @private */
192 userInput._checkShift = function( x,y ){
193 
194   var render = false;
195   var tpN = game.tileByPos(x,y);
196   var tpO = game.tileByPos(x+userInput._shiftOldX,y+userInput._shiftOldY);
197   if(
198     tpO !== tpN ||
199       screen.OVERLAYER[tpO] === true ||
200       screen.OVERLAYER[tpN] === true
201     ){
202     render = true;
203     if( screen.OVERLAYER[tpO] === true && y > 0 ){
204       screen.markForRedraw( x,y-1 );
205     }
206   }
207   else{
208 
209     var unitN = game.unitByPos(x,y);
210     var unitO = game.unitByPos(x+ userInput._shiftOldX,y+ userInput._shiftOldY);
211     if( unitN !== null ||unitO !== null ){
212       screen.markForRedrawWithNeighbours( x,y );
213     }
214     else{
215 
216       var propN = game.propertyByPos(x,y);
217       var propO = game.propertyByPos(x+ userInput._shiftOldX,y+ userInput._shiftOldY);
218       if( propN !== null ||propO !== null ){
219         render = true;
220         if( propO !== null && y > 0 ){
221           screen.markForRedraw( x,y-1 );
222         }
223       }
224     }
225   }
226 
227   if( render ) screen.markForRedraw( x,y );
228 };

230 /**
231  * Shifts the screen by a distance in a given direction. This
232  * function calculates the redraw map.
233  */
234 userInput.mapShift = function( dir, dis ){
235   if( dis === undefined ) dis = 1;
236 
237   // CHECK REDRAWS OF THE FOCUS TILES --> BEFORE
238   for(var x=screen.screenX,xe=screen.screenWidth; x<xe; x++){
239     for(var y=screen.screenY,ye=screen.screenHeight; y<ye; y++){
240       if( controller.valueOfSelectionPos(x,y) > 0 ){
241         screen.drawnMap[x-screen.screenX][y-screen.screenY] = true;
242         screen.drawChanges++;
243       }
244     }
245   }
246 
247   // UPDATE SCREEN META DATA
248   switch (dir) {
249 
250     case 0:
251       userInput._shiftOldX = 0;
252       userInput._shiftOldY = +dis;
253       screen.screenY = Math.max( 0, screen.screenY - dis );
254       break;
255 
256     case 1:
257       userInput._shiftOldX = -dis;
258       userInput._shiftOldY = 0;
259       screen.screenX = Math.min(
260         game.mapWidth()-screen.screenX-1, screen.screenX + dis
261       );
262       client.screenX = Math.max( 0, client.screenX );
263       break;
264 
265     case 2:
266       userInput._shiftOldX = 0;
267       userInput._shiftOldY = -dis;
268       screen.screenY = Math.min(
269         game.mapHeight()-screen.screenY-1, screen.screenY + dis
270       );
271       client.screenY = Math.max( 0, client.screenY );
272       break;
273 
274     case 3:
275       userInput._shiftOldX = +dis;
276       userInput._shiftOldY = 0;
277       screen.screenX = Math.max( 0, screen.screenX - dis );
278       break;
279   }
280 
281   // CHECK REDRAWS OF THE FOCUS TILES --> AFTER
282   for(var x=screen.screenX,xe=screen.screenWidth; x<xe; x++){
283     for(var y=screen.screenY,ye=screen.screenHeight; y<ye; y++){
284       if( controller.valueOfSelectionPos(x,y) > 0 ){
285         screen.drawnMap[x-screen.screenX][y-screen.screenY] = true;
286         screen.drawChanges++;
287       }
288 
289       userInput._checkShift( x,y );
290     }
291   }
292 
293   var x  = screen.screenX;
294   var yS = screen.screenY;
295   var xe = x+ screen.screenWidth;
296   var ye = yS+screen.screenHeight;
297 
298   // CHECK BOUNDS
299   if( xe > game.mapWidth() )  xe = game.mapWidth();
300   if( ye > game.mapHeight() ) ye = game.mapHeight();
301 
302   // ITERATE THROUGH THE SCREEN
303   for( ; x<xe; x++ ){
304     for( var y=yS ; y<ye; y++ ){
305 
306       screen.ammoStatusMap[x-screen.screenX][y-screen.screenY] = false;
307       screen.fuelStatusMap[x-screen.screenX][y-screen.screenY] = false;
308 
309       var unit = game.unitByPos(x,y);
310       if( unit !== null ){
311 
312         var us = game.unitSheet( unit.type );
313         if( us.maxAmmo > 0 &&
314           ( unit.ammo <= us.maxAmmo*0.35 || unit.ammo <= 1 ) ){
315 
316           screen.ammoStatusMap[x-screen.screenX][y-screen.screenY] = true;
317         }
318         if( unit.fuel <= us.maxFuel*0.25 || unit.fuel <= 5 ){
319           screen.fuelStatusMap[x-screen.screenX][y-screen.screenY] = true;
320         }
321       }
322     }
323   }
324 };

326 /**
327  * @private
328  */
329 userInput._setGeneratedPath = function(x,y){
330   var path = controller.returnPath(x,y);
331   userInput.resetCurrentMovePath( controller.selectedObject );
332   var cX = userInput.currentMovePathSourceX;
333   var cY = userInput.currentMovePathSourceY;
334   for( var i=0,e=path.length; i<e; i++ ){
335     userInput.currentMovePath[i] = path[i];
336     userInput.currentMoveIndex++;
337 
338     switch( path[i] ){
339       case game.MOVE_CODE_UP :    cY--; break;
340       case game.MOVE_CODE_RIGHT : cX++; break;
341       case game.MOVE_CODE_DOWN :  cY++; break;
342       case game.MOVE_CODE_LEFT :  cX--; break;
343     }
344 
345     userInput.currentMoveLeftPoints -=
346       controller.valueOfSelectionPos(cX,cY);
347   }
348 
349   userInput.currentMovePathX = cX;
350   userInput.currentMovePathY = cY;
351 
352   if( util.DEBUG ){
353     util.logInfo("set calculated path (",path,")");
354   }
355 };

357 /**
358  *
359  * @param x
360  * @param y
361  */
362 userInput.appendToCurrentMovePath = function( x,y ){
363   if( controller.valueOfSelectionPos(x,y) === -1 ||
364     ( x === userInput.currentMovePathX &&
365       y === userInput.currentMovePathY )
366      ||
367     ( x === userInput.currentMovePathSourceX &&
368       y === userInput.currentMovePathSourceY )
369     ) return;
370 
371   var diffX = Math.abs( userInput.currentMovePathX - x );
372   var diffY = Math.abs( userInput.currentMovePathY - y );
373 
374   if( util.DEBUG ) util.logInfo("diffX:",diffX," diffY:",diffY );
375 
376   if( diffX === 0 && diffY === 0 ){
377     // NOTHING HAPPENS
378     return;
379   }
380   else{
381     var diff = diffX+diffY;
382     if( diff === 1 ){
383 
384       var code;
385 
386       if( diffX === 1 && diffY === 0 ){
387         code = ( x < userInput.currentMovePathX )? game.MOVE_CODE_LEFT:
388                                                     game.MOVE_CODE_RIGHT;
389       }
390       else{
391         code = ( y < userInput.currentMovePathY )? game.MOVE_CODE_UP:
392                                                     game.MOVE_CODE_DOWN;
393       }
394 
395       var costs = controller.valueOfSelectionPos(x,y);
396       var cCode = userInput.currentMovePath[ userInput.currentMoveIndex-1 ];
397       if( userInput.currentMoveLeftPoints - costs >= 0 &&
398         ( userInput.currentMoveIndex === 0 || !(
399           ( cCode === game.MOVE_CODE_UP && code === game.MOVE_CODE_DOWN )
400             ||
401           ( cCode === game.MOVE_CODE_LEFT && code === game.MOVE_CODE_RIGHT )
402             ||
403           ( cCode === game.MOVE_CODE_DOWN && code === game.MOVE_CODE_UP )
404             ||
405           ( cCode === game.MOVE_CODE_RIGHT && code === game.MOVE_CODE_LEFT )
406           )
407         )
408       ){
409         userInput.currentMovePath[ userInput.currentMoveIndex ] = code;
410         userInput.currentMoveIndex++;
411         userInput.currentMoveLeftPoints = userInput.currentMoveLeftPoints - costs;
412         userInput.currentMovePathX = x;
413         userInput.currentMovePathY = y;
414 
415         if( util.DEBUG ){
416           util.logInfo("added (",x,",",y,") to the current path");
417         }
418       }
419       else{
420         if( util.DEBUG ){
421           util.logInfo(
422             "not enough points or back step, need to calculate path"
423           );
424         }
425 
426          userInput._setGeneratedPath(x,y);
427       }
428     }
429     else userInput._setGeneratedPath(x,y);
430 
431     if( util.DEBUG ){
432       util.logInfo(
433         "current move data",
434         "source (",userInput.currentMovePathSourceX,",",
435                    userInput.currentMovePathSourceY,")",
436         "target (",userInput.currentMovePathX,",",
437                    userInput.currentMovePathY,")",
438         "path (",  userInput.currentMovePath,")",
439         "rest points",userInput.currentMoveLeftPoints
440       );
441     }
442   }
443 };
