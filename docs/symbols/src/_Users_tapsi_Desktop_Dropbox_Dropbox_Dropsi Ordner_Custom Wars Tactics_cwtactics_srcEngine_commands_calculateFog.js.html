1 controller.engineAction({
  2 
  3   name:"calculateFog",
  4   
  5   key:"CCFO",
  6   
  7   /**
  8    * Calculates the fog map for a given player id.
  9    *
 10    * @param {Number} pid player id
 11    *
 12    * @methodOf controller.actions
 13    * @name calculateFog
 14    */
 15   action: function( pid ){
 16     var x;
 17     var y;
 18     var xe = model.mapWidth;
 19     var ye = model.mapHeight;
 20     var tid = model.players[pid].team;
 21     var fogEnabled = model.rules.fogEnabled;
 22     
 23     for( x=0 ;x<xe; x++ ){
 24       for( y=0 ;y<ye; y++ ){
 25         if( !fogEnabled ){
 26           model.fogData[x][y] = 1;
 27         }
 28         else{
 29           model.fogData[x][y] = 0;
 30         }
 31       }
 32     }
 33     
 34     if( fogEnabled ){
 35       for( x=0 ;x<xe; x++ ){
 36         for( y=0 ;y<ye; y++ ){
 37     
 38             //--------
 39             var unit = model.unitPosMap[x][y];
 40             if( unit !== null ){
 41               var sid = unit.owner;
 42               if( pid === sid || model.players[sid].team === tid ){
 43                 var vision = model.sheets.unitSheets[unit.type].vision;
 44                 controller.actions.addVision( x,y, vision );
 45               }
 46             }
 47       
 48             //--------
 49             var property = model.propertyPosMap[x][y];
 50             if( property !== null ){
 51               var sid = property.owner;
 52               if( pid === sid || model.players[sid].team === tid ){
 53                 var vision = model.sheets.tileSheets[property.type].vision;
 54                 controller.actions.addVision( x,y, vision );
 55               }
 56             }
 57         }
 58       }
 59     }
 60   }
 61   
 62 });
