1 controller.userAction({
  2 
  3   name:"join",
  4   key:"JNUN",
  5   
  6   unitAction: true,
  7 
  8   condition: function( mem ){
  9     var selectedUnit = mem.sourceUnit;
 10     var targetUnit = mem.targetUnit;
 11     
 12     if( selectedUnit === null || 
 13         targetUnit === null || 
 14         targetUnit.owner !== model.turnOwner ||
 15         targetUnit === selectedUnit ) return false;
 16     
 17     // NO LOAD MERGE
 18     if( model.hasLoadedIds( mem.sourceUnitId ) || 
 19         model.hasLoadedIds( mem.targetUnitId ) ) return false;
 20 
 21     return ( selectedUnit.type === targetUnit.type && targetUnit.hp < 89 );
 22   },
 23   
 24   createDataSet: function( mem ){
 25     return [ mem.sourceUnitId, mem.targetUnitId ];
 26   },
 27 
 28   /**
 29    * Joins an unit into an other.
 30    *
 31    * @param {Number} pid source unit id
 32    * @param {Number} tid target unit id
 33    *
 34    * @methodOf controller.actions
 35    * @name join
 36    */
 37   action: function( sid, tid ){
 38     var joinSource = model.units[sid];
 39     var joinTarget = model.units[tid];
 40     var junitSheet = model.sheets.unitSheets[ joinTarget.type ];
 41 
 42     // HEALTH POINTS
 43     controller.actions.healUnit( tid, joinSource.hp );
 44 
 45     // AMMO
 46     joinTarget.ammo += joinSource.ammo;
 47     if( joinTarget.ammo > junitSheet.maxAmmo ){
 48       joinTarget.ammo = junitSheet.maxAmmo;
 49     }
 50 
 51     // FUEL
 52     joinTarget.fuel += joinSource.fuel;
 53     if( joinTarget.fuel > junitSheet.maxFuel ){
 54       joinTarget.fuel = junitSheet.maxFuel;
 55     }
 56     
 57     // STATS
 58     var uSheet = model.sheets.unitSheets[ joinTarget.type ];
 59     var cFuel = joinTarget.fuel;
 60     var mFuel = uSheet.maxFuel;
 61     if( cFuel < parseInt(mFuel*0.25, 10) ) joinTarget.ST_LOW_FUEL = true;
 62     else                                   joinTarget.ST_LOW_FUEL = false;
 63     
 64     var cAmmo = joinTarget.ammo;
 65     var mAmmo = uSheet.maxAmmo;
 66     if( cAmmo <= parseInt(mAmmo*0.25, 10) ) joinTarget.ST_LOW_AMMO = true;
 67     else                                    joinTarget.ST_LOW_AMMO = false;
 68     if( mAmmo === 0 )                       joinTarget.ST_LOW_AMMO = false;
 69     
 70     var num = -1;
 71     if( joinTarget.hp <= 90 ){
 72       num = parseInt( joinTarget.hp/10 , 10 )+1;
 73     }
 74     joinTarget.ST_HP_NUM = num;
 75 
 76     controller.actions.destroyUnit( sid );
 77     controller.actions.wait( tid );
 78   }
 79 });
