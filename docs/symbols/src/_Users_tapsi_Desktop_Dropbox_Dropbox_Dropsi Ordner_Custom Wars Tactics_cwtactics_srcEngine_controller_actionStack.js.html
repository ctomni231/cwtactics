1 /** 
  2  * Action buffer object that holds all actions that aren't invoked yet.
  3  * 
  4  * @private 
  5  */
  6 controller.actionBuffer_ = util.createRingBuffer( CWT_ACTIONS_BUFFER_SIZE );
  7 
  8 /**
  9  * Pushes an action into the buffer and invokes a transaction process if the action
 10  * is marked as shared.
 11  * 
 12  * @example 
 13  *  data format
 14  *  [ parameters... , actionKey ]
 15  */
 16 controller.pushSharedAction = function(){
 17   var cmd = controller.actionObjects_[ arguments[arguments.length-1] ];
 18   if( cmd.shared && controller.isNetworkGame() ){
 19     this.sendNetworkMessage_( arguments );
 20   }
 21   
 22   controller.pushAction.apply( this, arguments );
 23 };
 24 
 25 /**
 26  * Pushes an action into the buffer.
 27  * 
 28  * @example 
 29  *  data format
 30  *  [ parameters... , actionKey ]
 31  */
 32 controller.pushAction = function(){
 33   if( DEBUG ){
 34     util.log("push action",JSON.stringify(arguments),"into action stack");
 35   }
 36 
 37   controller.actionBuffer_.push( arguments );
 38 };
 39 
 40 /**
 41  * Returns true if no action is in the action 
 42  * stack, else false.
 43  */
 44 controller.noNextActions  = function(){
 45   return controller.actionBuffer_.isEmpty();
 46 };
 47 
 48 /**
 49  * Pops the oldest action from the buffer and evaluates it. After the evaluation this 
 50  * function returns the action. If no action is in the buffer, null will be returned.
 51  */
 52 controller.doNextAction = function (){
 53   if( controller.actionBuffer_.isEmpty() ){
 54     return null;
 55   }
 56 
 57   var data = controller.actionBuffer_.pop();
 58   if( DEBUG ){
 59     util.log("evaluate action",JSON.stringify(data));
 60   }
 61 
 62   var cmd = controller.actionObjects_[ data[data.length-1] ];
 63   cmd.action.apply( cmd, data );
 64   
 65   return data;
 66 };
