149 /**
150  * Parses a data object into the database. The data object must be
151  * a valid javascript object. The type decide what kind of the the
152  * data object is.
153  */
154 model.parseSheet = function( data, type ){
155   var schema, db, excList;
156   var id = data.ID;
157   var validators = model.sheets.typeSheetValidators;
158 
159   // FIND SCHEMA AND DATA LIST
160   switch( type ){
161 
162     case model.sheets.UNIT_TYPE_SHEET:
163       schema =  validators.unitValidator;
164       db = model.sheets.unitSheets;
165       excList = validators.unitValidator.properties.ID.except;
166       break;
167 
168     case model.sheets.TILE_TYPE_SHEET:
169       schema =  validators.tileValidator;
170       db = model.sheets.tileSheets;
171       excList = validators.tileValidator.properties.ID.except;
172       break;
173 
174     case model.sheets.WEAPON_TYPE_SHEET:
175       schema =  validators.weaponValidator;
176       db = model.sheets.weaponSheets;
177       excList = validators.weaponValidator.properties.ID.except;
178       break;
179 
180     case model.sheets.WEATHER_TYPE_SHEET:
181       schema =  validators.weatherValidator;
182       db = model.sheets.weatherSheets;
183       excList = validators.weatherValidator.properties.ID.except;
184       break;
185 
186     case model.sheets.MOVE_TYPE_SHEET:
187       schema =  validators.movetypeValidator;
188       db = model.sheets.movetypeSheets;
189       excList = validators.movetypeValidator.properties.ID.except;
190       break;
191 
192     case model.sheets.RULESET:
193       schema =  validators.rulesValidator;
194       break;
195 
196     default: util.raiseError("unknow type",type);
197   }
198 
199   // CHECK IDENTICAL STRING FIRST
200   if( type !== model.sheets.RULESET &&
201     db.hasOwnProperty(id) ) util.raiseError(id,"is already registered");
202 
203   // VALIDATE SHEET
204   model.sheets._dbAmanda.validate( data, schema, function(e){
205     if( e ) util.raiseError( "failed to parse sheet due", e.getMessages() );
206   });
207 
208   if( type === model.sheets.RULESET ) model.sheets.defaultRules = data;
209   else{
210     db[id] = data;
211 
212     // REGISTER ID IN EXCEPTION LIST
213     excList.push(id);
214   }
215 
216 };

225 /**
226  * Returns all known type game of properties.
227  */
228 model.getListOfPropertyTypes = function(){
229   var tiles = model.sheets.tileSheets;
230   var l = Object.keys( tiles );
231   var r = [];
232   for( var i=l.length-1; i>=0; i-- ){
233     if( tiles[l[i]].capturePoints > 0 ){
234       r.push( l[i] );
235     }
236   }
237   return r;
238 };

255 /**
256  * Returns the primary weapon of an unit.
257  *
258  * @param unit unit object
259  */
260 model.primaryWeaponOfUnit = function( unit ){
261   if( DEBUG && unit === null ) util.raiseError();
262 
263   if( typeof unit === 'number' ){
264     unit = model.units[unit];
265   }
266 
267   var type = model.sheets.unitSheets[ unit.type ][ model.PRIMARY_WEAPON_TAG ];
268   return (type !== undefined )? model.sheets.weaponSheets[ type ] : null;
269 };

287 /**
288  * Returns the base damage from a weapon sheet against an unit type.
289  *
290  * @param weapon weapon sheet object
291  * @param {String} uType unit type
292  */
293 model.getBaseDamage = function( weapon, uType ){
294   if( DEBUG && weapon === null ) util.raiseError();
295   if( DEBUG && uType === null ) util.raiseError();
296 
297   var dmg;
298 
299   dmg = weapon.damages[ uType ];
300   if( dmg !== undefined ) return dmg;
301 
302   dmg = weapon.damages[ "*" ];
303   if( dmg !== undefined ) return dmg;
304 
305   return 0;
306 };
